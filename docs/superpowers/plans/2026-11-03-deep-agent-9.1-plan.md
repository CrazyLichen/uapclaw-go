> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 实现 DeepAgent 结构体，包装 ReActAgent + 任务循环 + Rails + 技能 + 子 Agent，作为面向生产的高级 Agent 封装层。

**Architecture:** 两层架构——外层 DeepAgent 负责多轮任务编排（任务循环、Rail 路由、状态管理、子 Agent 创建），内层 ReActAgent 负责单轮 Think-Act-Observe 推理。DeepAgent 同时实现 `DeepAgentInterface`（11 方法）和 `BaseAgent`（9 方法）。

**Tech Stack:** Go 1.22+, sync/atomic, context, channel-based streaming

// Configure 配置 DeepAgent。
// 首次调用走 initialConfigure，后续调用走 hotReconfigure。
// 对齐 Python: DeepAgent.configure(config) (line 150)
func (d *DeepAgent) Configure(ctx context.Context, config agentinterfaces.AgentConfig) error {
	cfg, ok := config.(*hschema.DeepAgentConfig)
	if !ok {
		return fmt.Errorf("config 类型应为 *DeepAgentConfig，实际 %T", config)
	}
	// 对齐 Python line 153: self._filter_disabled_tools(config)
	d.filterDisabledTools(cfg)

	d.configMu.Lock()
	if d.deepConfig == nil {
		// 对齐 Python line 154-155: self._initial_configure(config)
		d.initialConfigure(ctx, cfg)
	} else {
		// 对齐 Python line 156-157: self._hot_reconfigure(config)
		d.hotReconfigure(ctx, cfg)
	}
	d.configMu.Unlock()

	// 对齐 Python line 159: self._initialized = False
	d.initialized.Store(false)
	return nil
}
