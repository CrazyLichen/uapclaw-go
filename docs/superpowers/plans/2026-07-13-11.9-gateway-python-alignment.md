完整修改后的关键部分：

```go
// MessageHandlerInterface 消息处理器接口，由 MessageHandler 实现。
//
// 合并入站处理 + 出站消费为单一接口，对齐 Python ChannelManager(message_handler)。
type MessageHandlerInterface interface {
	// HandleMessage 处理入站消息（对齐 Python MessageHandler.handle_message）
	HandleMessage(msg *schema.Message)
	// ConsumeRobotMessages 从出站队列消费消息（对齐 Python MessageHandler.consume_robot_messages）
	ConsumeRobotMessages(timeout time.Duration) *schema.Message
}

// ChannelManager ...（字段中用 messageHandler 替代 inboundHandler + consumeProvider）
type ChannelManager struct {
	// channels 已注册的 Channel 实例映射（channelID → BaseChannel）
	channels map[string]BaseChannel
	// config Channel 相关配置（channelID → 配置 dict）
	config map[string]map[string]any
	// onConfigUpdated 配置更新回调
	onConfigUpdated OnConfigUpdatedFunc
	// pendingChannelRestart 待强制重启的 channelID 集合
	pendingChannelRestart map[string]struct{}
	// messageHandler 消息处理器（对齐 Python _message_handler）
	messageHandler MessageHandlerInterface
	// running 出站派发循环运行状态（对齐 Python _running）
	running atomic.Bool
	// dispatchCancel 出站派发循环取消函数（对齐 Python _dispatch_task）
	dispatchCancel context.CancelFunc
	// mu 保护 channels/config/pendingChannelRestart 的并发访问
	mu sync.RWMutex
}
```

func NewWebChannel(cfg WebChannelConfig, onConfigSaved OnConfigSavedFunc) *WebChannel {

		cfg.Port = defaultWebPort
	}
	if cfg.Path == "" {
		cfg.Path = defaultWSPath

	wc := &WebChannel{
		config:  cfg,
		clients: make(map[*websocket.Conn]bool),
		upgrader: websocket.Upgrader{
			CheckOrigin: wsorigin.GorillaCheckOrigin(),

		onConfigSavedCb: onConfigSaved,
	}

	// 创建事件推送回调
	sendEvent := func(event string, payload map[string]any) {
		wc.broadcastEvent(event, payload)
	}

	wc.dispatcher = RegisterWebHandlers(sendEvent, onConfigSaved)

	return wc
}

// NormalizeGatewayMessage 规范化 Gateway 消息。
//
// 对齐 Python: _normalize_gateway_message (app_gateway.py L91-122)：
//  1. params["query"] = params["content"]（如果 query 不存在且 content 存在）
//  2. CHAT_RESUME → CHAT_CANCEL + intent=resume
//  3. is_stream 推断
func NormalizeGatewayMessage(msg *schema.Message) *schema.Message {
	reqMethod := msg.ReqMethod
	if reqMethod == "" {
		reqMethod = schema.ReqMethodChatSend
	}

	// 解析 params
	var params map[string]any
	if len(msg.Params) > 0 {
		_ = json.Unmarshal(msg.Params, &params)
	}
	if params == nil {
		params = make(map[string]any)
	}

	// 1. content → query 映射
	if _, hasQuery := params["query"]; !hasQuery {
		if content, hasContent := params["content"]; hasContent {
			params["query"] = content
		}
	}

	// 2. CHAT_RESUME → CHAT_CANCEL + intent=resume
	if reqMethod == schema.ReqMethodChatResume {
		reqMethod = schema.ReqMethodChatCancel
		if _, hasIntent := params["intent"]; !hasIntent {
			params["intent"] = "resume"
		}
	}

	// 3. is_stream 推断
	methodVal := string(reqMethod)
	isStream := msg.IsStream || methodVal == string(schema.ReqMethodChatSend) || methodVal == string(schema.ReqMethodHistoryGet)

	// 重新编码 params
	paramsJSON, _ := json.Marshal(params)

	return &schema.Message{
		ID:        msg.ID,
		Type:      msg.Type,
		ChannelID: msg.ChannelID,
		SessionID: msg.SessionID,
		Params:    json.RawMessage(paramsJSON),
		Timestamp: msg.Timestamp,
		OK:        msg.OK,
		ReqMethod: reqMethod,
		Mode:      msg.Mode,
		IsStream:  isStream,
		StreamSeq: msg.StreamSeq,
		StreamID:  msg.StreamID,
		Provider:  msg.Provider,
	}
}

//
// 对齐 Python: _make_norm_and_forward (app_gateway.py L1007-1023)：
//  - method 在 forwardMethods 中 → normalize → deliverToMessageHandler
//  - method 在 noLocalMethods 中 → 返回 true（短路）
//  - 其他 → 返回 false（继续走本地 handler）
func MakeNormAndForward(
	channelMgr *cm.ChannelManager,
	forwardMethods map[string]bool,
	noLocalMethods map[string]bool,
	sourceLabel string,
) func(*schema.Message) bool {
	return func(msg *schema.Message) bool {
		methodVal := string(msg.ReqMethod)
		if !forwardMethods[methodVal] {
			return false
		}
		normalized := NormalizeGatewayMessage(msg)
		channelMgr.DeliverToMessageHandler(normalized)
		logger.Info(logComponentNormalize).
			Str("source", sourceLabel).
			Str("msg_id", msg.ID).
			Str("channel_id", msg.ChannelID).
			Msg("入站 -> MessageHandler")
		if noLocalMethods[methodVal] {
			return true
		}
		return false
	}
}
